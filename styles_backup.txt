    <script src="ml-engine.js"></script>
    <script>
        const mlEngine = new MLEngine();
        let isLoading = false;

        // Load model on page load
        async function initializeModel() {
            try {
                await mlEngine.loadModel();
                
                // Hide loading, show app
                document.getElementById('loading').style.display = 'none';
                document.getElementById('app').style.display = 'block';
                
                // Show model info
                const modelInfo = mlEngine.getModelInfo();
                const infoDiv = document.getElementById('modelInfo');
                const infoContent = document.getElementById('infoContent');
                
                infoContent.innerHTML = `
                    <div>Capas: ${modelInfo.layers}</div>
                    <div>Par치metros: ${modelInfo.parameters.toLocaleString()}</div>
                    <div>Rango de magnitud: [${modelInfo.magnitudeRange[0]}, ${modelInfo.magnitudeRange[1]}]</div>
                    <div>Rango de tama침o: [${modelInfo.sizeRange[0].toFixed(3)}, ${modelInfo.sizeRange[1].toFixed(3)}] km</div>
                `;
                infoDiv.style.display = 'block';
                
            } catch (error) {
                document.getElementById('loading').innerHTML = 
                    `<div class="result error">Error al cargar el modelo: ${error.message}</div>`;
            }
        }

        // Make prediction
        async function makePrediction() {
            if (isLoading) return;

            const magnitudeInput = document.getElementById('magnitude');
            const magnitude = parseFloat(magnitudeInput.value);
            const resultDiv = document.getElementById('result');
            const predictBtn = document.getElementById('predictBtn');

            // Validate input
            if (isNaN(magnitude)) {
                resultDiv.className = 'result error';
                resultDiv.textContent = 'Por favor ingresa un valor num칠rico v치lido';
                resultDiv.style.display = 'block';
                return;
            }

            // Show loading state
            isLoading = true;
            predictBtn.disabled = true;
            predictBtn.textContent = 'Prediciendo...';
            resultDiv.style.display = 'none';

            try {
                // Make prediction
                const result = await mlEngine.predict(magnitude);
                
                // Show result
                resultDiv.className = 'result success';
                resultDiv.innerHTML = `
                    <h3>Resultado de la Predicci칩n</h3>
                    <p><strong>Magnitud Absoluta (H):</strong> ${magnitude}</p>
                    <p><strong>Tama침o Estimado:</strong> ${result.size_km.toFixed(3)} km</p>
                    <p><strong>Di치metro:</strong> ${(result.size_km * 1000).toFixed(0)} metros</p>
                    <p style="font-size: 12px; color: #666; margin-top: 10px;">
                        Predicci칩n realizada a las ${result.timestamp.toLocaleTimeString()}
                    </p>
                `;
                resultDiv.style.display = 'block';

            } catch (error) {
                resultDiv.className = 'result error';
                resultDiv.textContent = 'Error en la predicci칩n: ' + error.message;
                resultDiv.style.display = 'block';
            } finally {
                isLoading = false;
                predictBtn.disabled = false;
                predictBtn.textContent = 'Predecir Tama침o';
            }
        }

        // Allow Enter key to submit
        document.addEventListener('DOMContentLoaded', () => {
            initializeModel();
            
            document.getElementById('magnitude').addEventListener('keypress', (e) => {
                if (e.key === 'Enter') {
                    makePrediction();
                }
            });
        });
    </script>
